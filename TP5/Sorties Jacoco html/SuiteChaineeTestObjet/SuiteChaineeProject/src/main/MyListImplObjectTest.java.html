package main;

import static org.junit.Assert.*;

import org.junit.Before;
import org.junit.Test;

public class MyListImplObjectTest {

	@Before
	public void setUp() throws Exception {
		
	}

	@Test
	public void testStart() {
		MyListImpl myList = new MyListImpl();
		
		assertTrue(myList.getStart() == null);
		
		myList.add(2);
		myList.add(3);
		myList.add(4);
		myList.add(5);
		myList.add(6);
		
		assertTrue(myList.getStart() != null);
		
		myList.removeAt(2);
		assertTrue(myList.getStart() != null);
		
		//au début
		myList.removeItem(2);
		assertTrue(myList.getStart() != null);
		
		//pas au debut
		myList.removeItem(5);
		assertTrue(myList.getStart() != null);

		//pas dedans
		try {
			myList.removeItem(0);
			fail("On attend un crash !!!!!!!!!!!!!");
		} catch (NullPointerException e) {
			assertTrue(myList.getStart() != null);
		}
		
		myList.setAt(18,1);
		assertTrue(myList.getStart() != null);
		
		myList.getAt(1);
		assertTrue(myList.getStart() != null);
		
		myList.reset();
		assertTrue(myList.getStart() == null);
		
		//Deuxième séquence
		myList = new MyListImpl();
		assertTrue(myList.getStart() == null);
		
		try {
			myList.removeItem(0);
			fail("On attend un crash !!!!!!!!!!!!!");
		} catch (NullPointerException e) {
			assertTrue(myList.getStart() == null);
		}
		
		myList.add(2);
		myList.add(3);
		myList.add(4);
		myList.add(5);
		myList.add(6);
		
		assertTrue(myList.getStart() != null);
		
		myList.removeAt(2);
		assertTrue(myList.getStart() != null);
		
		myList.setAt(18,1);
		assertTrue(myList.getStart() != null);
		
		myList.getAt(1);
		assertTrue(myList.getStart() != null);
		
		myList.reset();
		assertTrue(myList.getStart() == null);
	}
	
	@Test
	public void testSize() {
		MyListImpl myList = new MyListImpl();
		
		assertTrue(myList.getSize() == 0);
		
		myList.add(2);
		myList.add(3);
		myList.add(4);
		myList.add(5);
		myList.add(6);
		
		assertTrue(myList.getSize() > 0);
		
		myList.removeAt(0);
		assertTrue(myList.getSize() > 0);
		
		try {
			myList.removeAt(50);
			fail("On attend un crash !!!!!!!!!!!!!");
		} catch (ArrayIndexOutOfBoundsException e) {
			assertTrue(myList.getSize() > 0);
		}
		
		myList.removeItem(3);
		assertTrue(myList.getSize() > 0);
		
		try {
			myList.setAt(2,18);
			fail("On attend un crash !!!!!!!!!!!!!");
		} catch (ArrayIndexOutOfBoundsException e) {
			assertTrue(myList.getSize() > 0);
		}
		
		try {
			myList.getAt(42);
			fail("On attend un crash !!!!!!!!!!!!!");
		} catch (ArrayIndexOutOfBoundsException e) {
			assertTrue(myList.getSize() > 0);
		}
		
		myList.reset();
		assertTrue(myList.getSize() == 0);
	}
	
	@Test
	public void testCurrent() {
		MyListImpl myList = new MyListImpl();
		
		assertTrue(myList.getCurrent() == null);
		
		myList.add(2);
		myList.add(3);
		myList.add(4);
		myList.add(5);
		myList.add(6);
		
		assertTrue(myList.getCurrent() != null);
		
		myList.reset();
		assertTrue(myList.getCurrent() == null);
	}	

}
